;2) Escreva a definição de uma função que, a partir de dois números, escreva se são ou não múltiplos?

(define(multiplos x y) 
  (if(= (remainder x y) 0)
         "Multiplo"
         "Nao Multiplo"
     )
)

(multiplos 10 5)
(multiplos 9 2)
 
;3) Um supermercado deseja reajustar os preços de seus produtos, para mais ou
;para menos, de acordo com os critérios mostrados na tabela abaixo. Escrever
;uma definição de função que, a partir dos valores do preço atual e da venda
;média mensal do produto, calcule qual o preço reajustado?

(define (reajuste vmm pa) 
  (cond ((and (< vmm 500) (< pa 30)) (list (* pa 1.1) "Aumento 10%"))
        ((and (>= vmm 500) (<= vmm 1200) (>= pa 30) (< pa 80)) (list (* pa 1.15)"Aumento 15%"))
        ((and (>= vmm 1200) (<= pa 80)) (list (* pa 0.8) "Diminuição 20%"))
        )
)
 
 (reajuste 380 18)
 (reajuste 600 35)
 (reajuste 1300 80)
